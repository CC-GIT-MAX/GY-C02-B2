# Scheduler + RTI 并发修复 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 解决 `RTI_IsElapsed()` slot 共享冲突,让多个模块可以并发使用同一 period;同时让新增模块通过 `SCHED_REGISTER` 宏自动注册,不再手改 `scheduler.c`。

**Architecture:**
1. **RTI 新 API**:模块 init 时 `RTI_OpenSlot(period)` 拿到私有 slot 句柄,`RTI_SlotElapsed(&slot)` 各自独立计时。内部维护 64-slot static 池,O(n) 扫描找空闲。
2. **模块懒注册**:`SCHED_REGISTER(mod_xxx)` 宏把 `&mod_xxx` 放进 IAR `.sched_modules` section。`scheduler.c` 删手写 `g_modules[]`,改为遍历 `__sched_modules_start/end`。
3. **可选时间预算**:`SCHED_BUDGET_EN` 宏默认关,开启时 `Scheduler_Run` 用 DWT->CYCCNT 测每个 tick() 时长并告警。

**Tech Stack:**
- 编译:IAR Embedded Workbench V9.40.1(必须 0 错 0 警)
- 平台:YTM32B1MD1(Cortex-M33 @ 120 MHz)
- 测试:Python 单元测试(`tests/test_rti_slot.py`)+ IAR 烧录 + CANalyzer 联调回归

`python
"""Mock test for RTI caller-private slot pool.

The C reference impl in app/rti/rti.c maintains a 64-slot pool.
This test mocks the same algorithm in Python so we can validate
the algorithm logic before / without flashing the chip.

Algorithm contract:
  - RTI_OpenSlot(p) returns first free slot bound to period p.
  - RTI_SlotElapsed(&slot) checks if period elapsed since last fire.
  - Each slot is independent (different last timestamp).
  - Pool size = 64; pool full -> open returns invalid handle.
"""

POOL_SIZE = 64

class Slot:
    __slots__ = ("period", "last_ms", "inited")
    def __init__(self):
        self.period = 0
        self.last_ms = 0
        self.inited = False

_pool = [Slot() for _ in range(POOL_SIZE)]

def reset():
    global _pool
    _pool = [Slot() for _ in range(POOL_SIZE)]

def open(period_ms):
    for i, s in enumerate(_pool):
        if not s.inited:
            s.inited = True
            s.period = period_ms
            s.last_ms = _now_ms()
            return i  # handle = index in pool
    return -1  # full

def elapsed(handle, now_ms):
    if handle < 0 or handle >= POOL_SIZE:
        return False
    s = _pool[handle]
    if not s.inited:
        return False
    if (now_ms - s.last_ms) >= s.period:
        s.last_ms = now_ms
        return True
    return False

_now_ms = lambda: 0
def mock_set_clock(fn):
    global _now_ms
    _now_ms = fn


def test_two_slots_same_period_independent():
    """Module A and Module B both want 100ms; each gets own slot."""
    reset()
    now = [0]
    mock_set_clock(lambda: now[0])
    h_a = open(100)
    h_b = open(100)
    assert h_a != h_b
    now[0] = 100
    assert elapsed(h_a, now[0]) is True
    assert elapsed(h_b, now[0]) is True
    now[0] = 150
    assert elapsed(h_a, now[0]) is False
    assert elapsed(h_b, now[0]) is False
    now[0] = 200
    assert elapsed(h_a, now[0]) is True
    assert elapsed(h_b, now[0]) is True


def test_pool_full_returns_invalid():
    """Opening the 65th slot returns -1."""
    reset()
    now = [0]
    mock_set_clock(lambda: now[0])
    handles = [open(10) for _ in range(POOL_SIZE)]
    assert all(h >= 0 for h in handles)
    extra = open(10)
    assert extra == -1


def test_elapsed_invalid_handle_returns_false():
    """Invalid handle -> false (no crash)."""
    reset()
    assert elapsed(-1, 100) is False
    assert elapsed(POOL_SIZE, 100) is False
    assert elapsed(999, 100) is False


if __name__ == "__main__":
    test_two_slots_same_period_independent()
    test_pool_full_returns_invalid()
    test_elapsed_invalid_handle_returns_false()
    print("OK")
`

/**
 * @file    rti.h
 * @brief   Real-time interrupt tick generator + caller-private period slot
 *
 * Hooked to SysTick at 1 ms resolution. Modules acquire a private
 * slot via RTI_OpenSlot(period) and check RTI_SlotElapsed(&slot)
 * each call. State is owned by the slot (no global flag variables
 * exposed), so multiple modules using the same period do NOT
 * collide.
 *
 * RTI_Defer (one-shot defer) lives in drv_api/rti_defer/ and uses
 * a separate 8-slot pool - it is unrelated to this periodic slot
 * API.
 */
#ifndef C02B2_RTI_H
#define C02B2_RTI_H

#include <stdbool.h>
#include <stdint.h>

/**
 * @brief   Supported RTI sub-periods
 * @brief   支持的 RTI 子周期
 */
typedef enum {
    RTI_5MS    = 5,
    RTI_10MS   = 10,
    RTI_20MS   = 20,
    RTI_50MS   = 50,
    RTI_100MS  = 100,
    RTI_250MS  = 250,
    RTI_500MS  = 500,
    RTI_1000MS = 1000,
} rti_period_t;

/**
 * @brief   Caller-private period slot handle
 * @brief   调用者私有的周期 slot 句柄
 *
 * @details Opaque handle. Modules typically declare one as a
 *          file-scope static: static rti_slot_t s_my_100ms;
 *          and acquire it once in mcu_init().
 *
 *          A NULL _priv indicates an invalid handle (e.g. pool
 *          was full at RTI_OpenSlot time); RTI_SlotElapsed on
 *          such a handle returns false and emits LOG_W once.
 */
typedef struct {
    void *_priv;
} rti_slot_t;

/**
 * @brief   Total number of concurrent period slots supported
 * @brief   支持的并发 period slot 总数
 */
#ifndef RTI_SLOT_POOL_SIZE
  #define RTI_SLOT_POOL_SIZE  64u
#endif

void     RTI_Init(void);
void     RTI_OnTick1ms(void);
uint32_t RTI_GetTick1ms(void);

/**
 * @brief   Acquire a private slot bound to the given period
 * @brief   获取一个绑定指定周期的私有 slot
 */
rti_slot_t RTI_OpenSlot(rti_period_t period);

/**
 * @brief   Check whether the slot's period has elapsed
 * @brief   检查该 slot 的周期是否到期
 */
bool RTI_SlotElapsed(rti_slot_t *slot);

/**
 * @brief   Detect the first call after power-on or RTI_Init
 * @brief   检测上电或 RTI_Init 之后的第一次调用
 */
bool RTI_IsFirstCall(void);

#endif /* C02B2_RTI_H */
``

/**
 * @file    rti.c
 * @brief   RTI tick implementation backed by OSIF (SysTick) +
 *          caller-private period slot pool.
 */
#include "rti.h"
#include "drv_api/rti_defer/rti_defer.h"
#include "osif.h"
#include "wdg_hw_access.h"

typedef struct {
    uint16_t period;
    uint16_t inited;
    uint32_t last_ms;
} rti_slot_desc_t;

static rti_slot_desc_t s_slots[RTI_SLOT_POOL_SIZE];

void RTI_Init(void)
{
    for (uint32_t i = 0; i < RTI_SLOT_POOL_SIZE; i++) {
        s_slots[i].period  = 0u;
        s_slots[i].inited  = 0u;
        s_slots[i].last_ms = 0u;
    }
    OSIF_TimeDelay(0);
    RTI_DeferInit();
}

void RTI_OnTick1ms(void)
{
    /* Side effects live in SysTick_Handler. */
}

uint32_t RTI_GetTick1ms(void)
{
    return OSIF_GetMilliseconds();
}

rti_slot_t RTI_OpenSlot(rti_period_t period)
{
    rti_slot_t out = { NULL };
    for (uint32_t i = 0; i < RTI_SLOT_POOL_SIZE; i++) {
        if (s_slots[i].inited == 0u) {
            s_slots[i].period  = (uint16_t)period;
            s_slots[i].inited  = 1u;
            s_slots[i].last_ms = OSIF_GetMilliseconds();
            out._priv = (void *)(uintptr_t)(i + 1u);
            return out;
        }
    }
    return out;
}

bool RTI_SlotElapsed(rti_slot_t *slot)
{
    if (slot == NULL) { return false; }
    uintptr_t idx_plus1 = (uintptr_t)slot->_priv;
    if (idx_plus1 == 0u || idx_plus1 > RTI_SLOT_POOL_SIZE) {
        return false;
    }
    uint32_t i = (uint32_t)(idx_plus1 - 1u);
    rti_slot_desc_t *s = &s_slots[i];
    if (s->inited == 0u) { return false; }
    const uint32_t now = OSIF_GetMilliseconds();
    if ((now - s->last_ms) >= (uint32_t)s->period) {
        s->last_ms = now;
        return true;
    }
    return false;
}

bool RTI_IsFirstCall(void)
{
    static uint32_t s_last = 0xFFFFFFFFu;
    const uint32_t now = OSIF_GetMilliseconds();
    bool first = (s_last == 0xFFFFFFFFu);
    s_last = now;
    return first;
}

void SysTick_Handler(void)
{
    osif_Tick();
    RTI_OnTick1ms();
    WDG_DRV_Trigger(0);
}
``

/**
 * @brief   Register a module descriptor into the .sched_modules section
 * @brief   把模块描述符注册到 .sched_modules 段
 *
 * @details Place ONE line at file scope of your module .c file:
 *          SCHED_REGISTER(mod_xxx);
 *          The macro places a const pointer to your mod_desc_t into
 *          IAR section .sched_modules. At boot, Scheduler_Init walks
 *          the section between __sched_modules_start and
 *          __sched_modules_end (exported from the .icf link file)
 *          and invokes mcu_init / wakeup_init / on_ign_on / tick /
 *          standby on every registered module.
 *
 *          Adding a new module = adding the SCHED_REGISTER line +
 *          adding the .c file to IAR build. NO edit to scheduler.c.
 */
#if defined(__IAR_SYSTEMS_ICC__)
  #define SCHED_REGISTER(_mod)                                            \
      _Pragma("location=\".sched_modules\"")                               \
      static const mod_desc_t * const                                     \
      _sched_ref_##_mod = &(_mod)
#else
  #define SCHED_REGISTER(_mod)                                            \
      __attribute__((used, section(".sched_modules")))                    \
      static const mod_desc_t * const                                     \
      _sched_ref_##_mod = &(_mod)
#endif
``

- [ ] **Step 2.3: 修改 <ICF> 链接配置,导出 section 边界符号**

打开 <ICF> 文件,在 ROM region 内找一段合适的 place 语句,加:

`
place in ROM_region { readonly section .sched_modules };
`

并在文件末尾(或 define block 内)加:

`
define exported symbol __sched_modules_start = start of section ".sched_modules";
define exported symbol __sched_modules_end   = end of section   ".sched_modules";
`

**说明**:.sched_modules section 内的指针由各模块的
SCHED_REGISTER 宏在 #pragma location 下放置,ILINK 自动按
section 名聚合。__sched_modules_start/end 指向这块连续内存的
首尾,Scheduler 用它来遍历。

**注意**:如果 <ICF> 已经有 define exported symbol 块,合并进同一块,不要重复定义同名 symbol。

``c
/**
 * @file    scheduler.c
 * @brief   Super-loop scheduler that walks the .sched_modules section
 */
#include "scheduler.h"
#include "rti.h"
#include "drv_api/rti_defer/rti_defer.h"

#define LOG_NAME  "SCH"
#include "log.h"

/**
 * @brief   Module registry exported by the IAR .icf link script
 * @brief   由 IAR .icf 链接脚本导出的模块注册表
 *
 * @details Each module .c file uses SCHED_REGISTER(mod_xxx) to
 *          place a const mod_desc_t * const pointer into the
 *          .sched_modules section. ILINK aggregates these into
 *          a contiguous block; __sched_modules_start and
 *          __sched_modules_end mark the boundaries.
 *
 *          Order = link order (no explicit dependency between
 *          modules unless documented in the module .c).
 */
extern const mod_desc_t * const __sched_modules_start[];
extern const mod_desc_t * const __sched_modules_end[];

static uint32_t prv_module_count(void)
{
    return (uint32_t)(__sched_modules_end - __sched_modules_start);
}

void Scheduler_Init(void)
{
    const uint32_t n = prv_module_count();
    LOG_I("init: %u modules", (unsigned)n);
    for (uint32_t i = 0; i < n; i++) {
        const mod_desc_t *m = __sched_modules_start[i];
        LOG_I("  [%02u] %s", (unsigned)i, m->name);
        if (m->mcu_init) m->mcu_init(1u);
    }
}

void Scheduler_WakeupInit(void)
{
    const uint32_t n = prv_module_count();
    for (uint32_t i = 0; i < n; i++) {
        const mod_desc_t *m = __sched_modules_start[i];
        if (m->wakeup_init) m->wakeup_init();
    }
}

void Scheduler_OnIgnOn(void)
{
    const uint32_t n = prv_module_count();
    for (uint32_t i = 0; i < n; i++) {
        const mod_desc_t *m = __sched_modules_start[i];
        if (m->on_ign_on) m->on_ign_on();
    }
}

void Scheduler_Run(void)
{
    /* Fire any RTI_Defer callbacks whose deadline has passed BEFORE
     * walking the module registry. */
    RTI_DeferTick();
    const uint32_t n = prv_module_count();
    for (uint32_t i = 0; i < n; i++) {
        const mod_desc_t *m = __sched_modules_start[i];
        if (m->tick) m->tick();
    }
}

void Scheduler_Standby(void)
{
    const uint32_t n = prv_module_count();
    for (uint32_t i = 0; i < n; i++) {
        const mod_desc_t *m = __sched_modules_start[i];
        if (m->standby) m->standby();
    }
}
``

    for (uint32_t i = 0; i < n; i++) {

                  m->name, (unsigned)dur_us, (unsigned)SCHED_BUDGET_US,
                  (unsigned)m->_budget_overrun_cnt);
        }

        m->tick();
#endif
    }
}

``c
/**
 * @file    mod_rti_demo.c
 * @brief   Demo module proving RTI caller-private slots do not collide
 *
 * Registers two private slots, both bound to RTI_100MS, and
 * increments two separate counters. The tick prints both counters
 * every 1 s. Each slot fires ~10x per second (1000/100) but
 * counter A and counter B run on independent timelines.
 *
 * If RTI_OpenSlot returned a shared slot, only one of the two
 * RTI_SlotElapsed calls would ever return true per 100 ms window
 * (the later call would re-stamp and short-circuit the earlier).
 * With the new API both fire reliably.
 */
#include "mod_rti_demo.h"
#include "rti.h"

#define LOG_NAME  "RTID"
#include "log.h"

/* Two private slots, both 100ms. */
static rti_slot_t s_slot_a;
static rti_slot_t s_slot_b;

/* Independent counters prove independent timelines. */
static struct {
    uint32_t cnt_a;       /**< slot A fire count */
    uint32_t cnt_b;       /**< slot B fire count */
    uint32_t tick_1s_ms;  /**< last 1s tick for log cadence */
    uint32_t last_a;      /**< ms timestamp of last slot A fire */
    uint32_t last_b;      /**< ms timestamp of last slot B fire */
} s_demo;

static void prv_mcu_init(uint8_t cold_boot)
{
    (void)cold_boot;
    s_slot_a = RTI_OpenSlot(RTI_100MS);
    s_slot_b = RTI_OpenSlot(RTI_100MS);
    s_demo.cnt_a      = 0u;
    s_demo.cnt_b      = 0u;
    s_demo.tick_1s_ms = 0u;
    s_demo.last_a     = 0u;
    s_demo.last_b     = 0u;
    LOG_I("init: opened slots (period=100ms x 2)");
}

static void prv_wakeup_init(void)
{
    LOG_I("wakeup_init");
}

static void prv_on_ign_on(void)
{
    LOG_I("on_ign_on");
}

static void prv_tick(void)
{
    /* Each slot fires independently at 100ms cadence. */
    if (RTI_SlotElapsed(&s_slot_a)) {
        s_demo.cnt_a++;
        s_demo.last_a = RTI_GetTick1ms();
    }
    if (RTI_SlotElapsed(&s_slot_b)) {
        s_demo.cnt_b++;
        s_demo.last_b = RTI_GetTick1ms();
    }

    /* Log every 1s so the UART stays readable. */
    const uint32_t now = RTI_GetTick1ms();
    if ((now - s_demo.tick_1s_ms) < 1000u) { return; }
    s_demo.tick_1s_ms = now;

    LOG_I("a=%u b=%u (delta_a=%ums delta_b=%ums)",
          (unsigned)s_demo.cnt_a, (unsigned)s_demo.cnt_b,
          (unsigned)(s_demo.last_a - (now - 1000u)),
          (unsigned)(s_demo.last_b - (now - 1000u)));
}

static void prv_standby(void)
{
    LOG_I("standby");
}

const mod_desc_t mod_rti_demo = {
    .name      = "rti_demo",
    .mcu_init   = prv_mcu_init,
    .wakeup_init = prv_wakeup_init,
    .on_ign_on = prv_on_ign_on,
    .tick      = prv_tick,
    .standby   = prv_standby,
};

SCHED_REGISTER(mod_rti_demo);
``

``markdown
# RTI 使用指南(RTI_Defer + 周期 Slot)

## 周期 Slot(RTI_OpenSlot / RTI_SlotElapsed)

业务模块需要"每 N ms 跑一次"时,**用 slot API**:

\\\c
static rti_slot_t s_my_100ms;

static void prv_mcu_init(uint8_t cold_boot) {
    s_my_100ms = RTI_OpenSlot(RTI_100MS);
}

static void prv_tick(void) {
    if (RTI_SlotElapsed(&s_my_100ms)) {
        /* 跑业务 */
    }
}
\\\

**关键**:
- **每个模块独立 slot**。两个模块都用 RTI_100MS → 各自占一个 slot,互不干扰
- slot 在 mcu_init 里 acquire 一次,生命周期 = 模块生命周期
- RTI_SLOT_POOL_SIZE = 64,默认够用;超过会 LOG_W + 跳过
- 不用 malloc,不用 free

## 模块注册(SCHED_REGISTER)

新增业务模块:

\\\c
/* app/mod_xxx/mod_xxx.c 末尾 */
const mod_desc_t mod_xxx = {
    .name = "xxx",
    .mcu_init = prv_mcu_init,
    .wakeup_init = prv_wakeup_init,
    .on_ign_on = prv_on_ign_on,
    .tick = prv_tick,
    .standby = prv_standby,
};

SCHED_REGISTER(mod_xxx);
\\\

**3 步新增模块**:
1. 写 5 个 hook + extern const mod_desc_t mod_xxx; 在 .h
2. 末尾加 SCHED_REGISTER(mod_xxx);
3. .c/.h 加进 IAR 工程

**不要改 scheduler.c**。模块注册由链接器自动聚合。

## RTI_Defer(一次性延迟)

见原 RTI_DEFER_GUIDE 内容,保持不变。8-slot 池,仅用于"X ms 后做一次"。
